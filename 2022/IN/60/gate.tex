\begin{align}
V_o &=\abs{\frac{X_c}{R + X_c}}\cdot V_i \\
8V_p &=\abs{\frac{\frac{1}{j\omega C}}{R + \frac{1}{j\omega C}}}\cdot 10V_p \\
\frac{4}{5} &= \abs{\frac{1}{1 + j\omega RC}} \\
\frac{16}{25} &= \frac{1}{1 + \omega^2R^2C^2} \\
\omega^2R^2C^2 &= \frac{9}{16} \\
\omega RC &= \frac{3}{4} \\
C &= \frac{3}{4\cdot 50 \cdot 2\pi\cdot 10^3} \\
\implies C &= 2.387  \mu F
\end{align}
